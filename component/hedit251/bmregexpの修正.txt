function TREScanner.GetToken: REToken_t;
begin
  if FInCharClass then begin
    try
      result := GetTokenCC;
    except
      FInCharClass := False;
      raise;
    end;
    if result = retk_RBra then
      FInCharClass := False;
  end else begin
    result := GetTokenStd;
    if (result = retk_LBra) or (result = retk_LBraNeg) then
      FInCharClass := True;
  end;
end;
